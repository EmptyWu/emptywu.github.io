## 進階題，不適合新手
### 字串切割+if
* 如果第一個數字是質數，請將第二、三個數字加總
* 如果第一個數字是偶數，請將第二、三個數字相減
* 如果第一個數字不是質數也不是偶數，請將第二、三個數字相乘
```JavaScript
function isPrime(num){
    if(num >=100)
    {
         return false;
    }
    else if(num >2){
        const t = parseInt(Math.sqrt(num));
        for (let i = 2; i <= t; i++) {
            if (num % i == 0) {
                return false;
            }
        }
        return true;
    }else {
        return false;
    }
}
function sum(data){
    const num=data.split(',');
    if(isPrime(num[0])){
        //質數
        return parseInt(num[1])+parseInt(num[2]);
    }
    else if(num[0]%2===0){
        //偶數
        return parseInt(num[1])-parseInt(num[2]);
    }
    else
    {
        return parseInt(num[1])*parseInt(num[2]);
    }
    
}
// input 輸入
sum("13,20,10")
sum("10,2,3")
sum("9,5,7")
// output 輸出
30
-1
35
```
