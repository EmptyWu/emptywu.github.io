### 題目三：滿 300 即可符合優惠條件，能使用 50 元折價券

```JavaScript
function buy(num){
    if(num >=400)
        return `符合優惠券資格，費用是 ${num-50}`;
    else if(num >=300 && num <400)
        return `剛好符合資格，費用是 ${num-50}`;
    else if(num >0 && num <300)
        return `不優惠券資格，費用是 ${num}`;
    else if( num ===0)
        return `...你來亂的嗎？`;
    else if( num <0)
        return `再鬧我就生氣了哦~`;
}
// input 輸入
buy(450)
buy(299)
buy(0)
buy(-30)
buy(300)
buy(1000)

// output 輸出
"符合優惠券資格，費用是 400"

"...你來亂的嗎？"
"再鬧我就生氣了哦~"
"剛好符合資格，費用是 250"
"符合優惠券資格，費用是 950"
```

## 進階題，不適合新手
### 是否為質數
```JavaScript
function isPrime(num){
    if(num >=100)
    {
         return "我只能算 100 以內的質數計算，原諒我QQ";
    }
    else if(num >2){
        const t = parseInt(Math.sqrt(num));
        for (let i = 2; i <= t; i++) {
            if (num % i == 0) {
                return "不是質數";
            }
        }
        return "是質數";
    }else {
        return "無法判別";
    }
}
// input 輸入
isPrime(29)
isPrime(17)
isPrime(-3)
isPrime(288)
isPrime(97)
isPrime(6)
// output 輸出
"是質數"
"是質數"
"無法判別"
"我只能算 100 以內的質數計算，原諒我QQ"
"是質數"
"不是質數"
```
